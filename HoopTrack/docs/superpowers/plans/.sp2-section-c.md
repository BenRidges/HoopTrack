```swift
// KillfeedView.swift
// SP2: Scrolling shot log that fades out entries older than killfeedFadeOutSec.
// Relies on HoopTrack.Game.killfeedMaxEntries and killfeedFadeOutSec constants
// added in Section A.

import SwiftUI

struct KillfeedView: View {
    @Binding var entries: [KillfeedEntry]

    var body: some View {
        TimelineView(.animation) { timeline in
            let now = timeline.date
            VStack(alignment: .trailing, spacing: 4) {
                ForEach(visibleEntries) { entry in
                    let age = now.timeIntervalSince(entry.timestamp)
                    let opacity = entryOpacity(age: age)
                    killfeedRow(entry: entry)
                        .opacity(opacity)
                        .animation(.easeOut(duration: 0.3), value: opacity)
                }
            }
        }
    }

    // MARK: - Helpers

    private var visibleEntries: [KillfeedEntry] {
        Array(entries.prefix(HoopTrack.Game.killfeedMaxEntries))
    }

    private func entryOpacity(age: TimeInterval) -> Double {
        let fadeStart = HoopTrack.Game.killfeedFadeOutSec * 0.7
        let fadeEnd   = HoopTrack.Game.killfeedFadeOutSec
        guard age < fadeEnd else { return 0 }
        guard age > fadeStart else { return 1 }
        return 1 - (age - fadeStart) / (fadeEnd - fadeStart)
    }

    private func teamColour(for team: TeamAssignment) -> Color {
        switch team {
        case .teamA: return .orange
        case .teamB: return Color(red: 0.2, green: 0.5, blue: 1.0)
        }
    }

    private func resultSymbol(for result: ShotResult) -> String {
        result == .make ? "✓" : "✗"
    }

    private func killfeedRow(_ entry: KillfeedEntry) -> some View {
        HStack(spacing: 6) {
            Text("\(entry.playerName) — \(entry.shotType.rawValue) \(resultSymbol(for: entry.result))")
                .font(.system(size: 13, weight: .semibold))
                .foregroundStyle(teamColour(for: entry.teamAssignment))
        }
        .padding(.horizontal, 10)
        .padding(.vertical, 5)
        .background(.black.opacity(0.55), in: RoundedRectangle(cornerRadius: 8))
    }
}
```

```swift
// PlayerOverlay.swift
// SP2: Name chips drawn at normalised player bounding box positions.
// Accepts a plain value-type array so the view doesn't import PlayerTracker.

import SwiftUI

/// Display record for one tracked player. SP2's PlayerTracker creates these
/// before passing to the overlay — keeping the view free of service imports.
struct TrackedPlayerDisplay {
    let name: String
    let teamAssignment: TeamAssignment
    /// Normalised bounding box in camera-coordinate space (0–1 each axis).
    let normalizedRect: CGRect
    let matchConfidence: Double
}

struct PlayerOverlay: View {
    let players: [TrackedPlayerDisplay]

    var body: some View {
        GeometryReader { geo in
            ForEach(players.indices, id: \.self) { idx in
                let player = players[idx]
                guard player.matchConfidence > HoopTrack.Game.playerOverlayMinConfidence else {
                    return AnyView(EmptyView())
                }
                let screenRect = screenRect(from: player.normalizedRect, in: geo.size)
                let chipCenter = CGPoint(
                    x: screenRect.midX,
                    y: screenRect.minY - 14   // float chip above top edge
                )
                return AnyView(
                    nameChip(player: player)
                        .position(chipCenter)
                )
            }
        }
    }

    // MARK: - Helpers

    private func screenRect(from norm: CGRect, in size: CGSize) -> CGRect {
        CGRect(
            x:      norm.minX * size.width,
            y:      norm.minY * size.height,
            width:  norm.width  * size.width,
            height: norm.height * size.height
        )
    }

    private func teamColour(for team: TeamAssignment) -> Color {
        switch team {
        case .teamA: return .orange
        case .teamB: return Color(red: 0.2, green: 0.5, blue: 1.0)
        }
    }

    private func nameChip(player: TrackedPlayerDisplay) -> some View {
        Text(player.name)
            .font(.system(size: 11, weight: .bold))
            .foregroundStyle(.white)
            .padding(.horizontal, 8)
            .padding(.vertical, 4)
            .background(
                Capsule()
                    .fill(.black.opacity(0.65))
                    .overlay(
                        Capsule()
                            .strokeBorder(teamColour(for: player.teamAssignment), lineWidth: 1.5)
                    )
            )
    }
}
```

```swift
// GameSummaryView.swift
// SP2: Full box score — hero score, team totals, per-player table,
// shot chart, and unresolved-shot link.

import SwiftUI
import SwiftData

struct GameSummaryView: View {
    let session: GameSession
    let onDone: () -> Void

    @EnvironmentObject private var dataService: DataService

    var body: some View {
        NavigationStack {
            ScrollView {
                VStack(spacing: 20) {
                    heroSection
                    teamTotalsSection
                    playerTableSection
                    shotChartSection
                    unresolvedSection
                }
                .padding()
            }
            .navigationTitle("Game Summary")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .confirmationAction) {
                    Button("Done", action: onDone).bold().tint(.orange)
                }
            }
        }
    }

    // MARK: - Hero

    private var heroSection: some View {
        VStack(spacing: 8) {
            HStack(spacing: 0) {
                teamBlock(label: "TEAM A", score: session.teamAScore, colour: .orange)
                Text("—")
                    .font(.system(size: 32, weight: .black))
                    .foregroundStyle(.secondary)
                    .frame(width: 48)
                teamBlock(label: "TEAM B", score: session.teamBScore,
                          colour: Color(red: 0.2, green: 0.5, blue: 1.0))
            }

            // Duration
            Label(durationString, systemImage: "clock")
                .font(.subheadline)
                .foregroundStyle(.secondary)

            // MVP chip
            if let mvp = mvpPlayer {
                HStack(spacing: 6) {
                    Image(systemName: "star.fill").foregroundStyle(.yellow)
                    Text("MVP: \(mvp.name)")
                        .font(.subheadline.bold())
                }
                .padding(.horizontal, 14)
                .padding(.vertical, 6)
                .background(.yellow.opacity(0.15), in: Capsule())
            }
        }
        .padding()
        .background(.ultraThinMaterial, in: RoundedRectangle(cornerRadius: 16))
    }

    private func teamBlock(label: String, score: Int, colour: Color) -> some View {
        VStack(spacing: 2) {
            Text(label)
                .font(.system(size: 10, weight: .semibold))
                .tracking(1.2)
                .foregroundStyle(colour)
            Text("\(score)")
                .font(.system(size: 52, weight: .black, design: .rounded))
                .monospacedDigit()
        }
        .frame(maxWidth: .infinity)
        .padding(.vertical, 8)
    }

    // MARK: - Team Totals

    private var teamTotalsSection: some View {
        VStack(alignment: .leading, spacing: 10) {
            Text("Team Totals")
                .font(.headline)

            HStack {
                teamTotalsColumn(team: .teamA, colour: .orange)
                Divider()
                teamTotalsColumn(team: .teamB,

    private func teamTotalsColumn(team: TeamAssignment, colour: Color) -> some View {
        let shots = session.shots.filter { $0.shooter?.teamAssignment == team }
        let fga = shots.count
        let fgm = shots.filter { $0.result == .make }.count
        let fgPct = fga > 0 ? Double(fgm) / Double(fga) * 100 : 0

                .foregroundStyle(colour)

            statRow(label: "3PA / 3PM", value: "\(tpa) / \(tpm)")

            Text(label).font(.caption).foregroundStyle(.secondary)

    // MARK: - Player Table

    private var playerTableSection: some View {

            ForEach(session.players) { player in

                let fgPct  = stats.fga > 0 ? Double(stats.fgm) / Double(stats.fga) * 100 : 0
                let tpPct  = stats.tpa > 0 ? Double(stats.tpm) / Double(stats.tpa) * 100 : 0

                    .frame(maxWidth: .infinity, alignment: .leading)

                    Text("\(stats.pts)").font(.subheadline.bold()).frame(width: 32, alignment: .trailing)
                    Text("\(stats.fga)").font(.caption).frame(width: 36, alignment: .trailing)
                    Text(String(format: "%.0f%%", fgPct)).font(.caption).frame(width: 40, alignment: .trailing)
                    Text("\(stats.tpa)").font(.caption).frame(width: 36, alignment: .trailing)

                .padding(.vertical, 4)

                Divider()
            }
        }
        .padding()
        .background(.ultraThinMaterial, in: RoundedRectangle(cornerRadius: 12))
    }

    // MARK: - Shot Chart

    private var shotChartSection: some View {
        VStack(alignment: .leading, spacing: 8) {
            Text("Shot Chart")
                .font(.headline)

            // CourtMapView takes [ShotRecord] (solo session model).
            // Convert via a lightweight shim so no game model is mutated.
            CourtMapView(shots: gameShotsAsShotRecords())

            CourtMapLegend()
        }
        .padding()
        .background(.ultraThinMaterial, in: RoundedRectangle(cornerRadius: 12))
    }

    // MARK: - Unresolved shots

    @ViewBuilder
    private var unresolvedSection: some View {
        let unresolved = session.shots.filter { $0.shooter == nil }
        if !unresolved.isEmpty {
            NavigationLink {
                UnresolvedShotsView(session: session)
                    .environmentObject(dataService)
            } label: {
                HStack {
                    Image(systemName: "questionmark.circle.fill")
                        .foregroundStyle(.orange)
                    Text("Resolve \(unresolved.count) unattributed shot\(unresolved.count == 1 ? "" : "s")")
                        .font(.subheadline.bold())
                        .foregroundStyle(.primary)
                    Spacer()
                    Image(systemName: "chevron.right")
                        .foregroundStyle(.secondary)
                }
                .padding()
                .background(.ultraThinMaterial, in: RoundedRectangle(cornerRadius: 12))
            }
        }
    }

    // MARK: - Computed helpers

    private var durationString: String {
        let mins = Int(session.durationSeconds) / 60
        let secs = Int(session.durationSeconds) % 60
        return mins > 0 ? "\(mins)m \(secs)s" : "\(secs)s"
    }

    private var mvpPlayer: GamePlayer? {
        session.players.max { a, b in
            playerStats(for: a).pts < playerStats(for: b).pts
        }
    }

    /// Per-player box score stats. Handles the 3PT-also-counts-in-fgm accounting.
    private func playerStats(for player: GamePlayer) -> (pts: Int, fga: Int, fgm: Int, tpa: Int, tpm: Int, avgConf: Double) {
        let playerShots = session.shots.filter { $0.shooter?.id == player.id }
        let fga = playerShots.count
        let fgm = playerShots.filter { $0.result == .make }.count
        let threes = playerShots.filter { $0.shotType == .threePoint }
        let tpa = threes.count
        let tpm = threes.filter { $0.result == .make }.count
        // 2PT makes × 2 pts + 3PT makes × 3 pts (3PT makes are subset of fgm)
        let pts = (fgm - tpm) * 2 + tpm * 3
        let avgConf = playerShots.isEmpty
            ? 0
            : playerShots.map { $0.attributionConfidence }.reduce(0, +) / Double(playerShots.count)
        return (pts, fga, fgm, tpa, tpm, avgConf)
    }

    /// Shim: build lightweight ShotRecord stand-ins from GameShotRecord so
    /// CourtMapView can render game shots without a real SwiftData context.
    private func gameShotsAsShotRecords() -> [ShotRecord] {
        session.shots.compactMap { gameShot in
            // ShotRecord init — uses the existing @Model type; we create
            // unmanaged instances purely for display (never inserted into context).
            let sr = ShotRecord(
                courtX: gameShot.courtX,
                courtY: gameShot.courtY,
                result: gameShot.result,
                shotType: .catchAndShoot,   // GameShotType has no ShotType equivalent; filler
                releaseAngle: nil,
                jumpHeightCm: nil,
                releaseDurationMs: nil,
                session: nil
            )
            return sr
        }
    }
}
```

```swift
// UnresolvedShotsView.swift
// SP2: Post-game manual attribution for shots where PlayerTracker confidence
// fell below the minimum threshold (shooter == nil on GameShotRecord).
// Presented as a NavigationLink from GameSummaryView.

import SwiftUI
import SwiftData

struct UnresolvedShotsView: View {
    let session: GameSession
    @EnvironmentObject private var dataService: DataService
    @Environment(\.dismiss) private var dismiss

    @State private var unresolvedShots: [GameShotRecord] = []

    var body: some View {
        VStack(spacing: 0) {
            if unresolvedShots.isEmpty {
                allResolvedPlaceholder
            } else {
                headerBar
                shotCarousel
                Spacer()
                progressIndicator
            }
        }
        .navigationTitle("Unresolved Shots")
        .navigationBarTitleDisplayMode(.inline)
        .onAppear { loadUnresolved() }
    }

    // MARK: - Subviews

    private var allResolvedPlaceholder: some View {
        VStack(spacing: 16) {
            Image(systemName: "checkmark.seal.fill")
                .font(.system(size: 52))
                .foregroundStyle(.green)
            Text("All shots attributed")
                .font(.headline)
            Text("Every shot has been assigned to a player.")
                .font(.subheadline)
                .foregroundStyle(.secondary)
                .multilineTextAlignment(.center)
            Button("Done") { dismiss() }
                .buttonStyle(.borderedProminent)
                .tint(.orange)
        }
        .padding(40)
    }

    private var headerBar: some View {
        HStack {
            Text("\(unresolvedShots.count) shot\(unresolvedShots.count == 1 ? "" : "s") to assign")
                .font(.subheadline)
                .foregroundStyle(.secondary)
            Spacer()
            Button("Skip All") {
                dismiss()
            }
            .font(.subheadline)
            .tint(.secondary)
        }
        .padding(.horizontal)
        .padding(.top, 12)
        .padding(.bottom, 4)
    }

    private var shotCarousel: some View {
        ScrollView(.horizontal, showsIndicators: false) {
            HStack(alignment: .top, spacing: 16) {
                ForEach(unresolvedShots) { shot in
                    ShotAttributionCard(
                        shot: shot,
                        players: session.players,
                        onAssign: { player in
                            assignShot(shot, to: player)
                        }
                    )
                }
            }
            .padding(.horizontal, 20)
            .padding(.vertical, 16)
        }
    }

    private var progressIndicator: some View {
        let total = (session.shots.filter { $0.shooter == nil }.count) +
                    (session.shots.count - unresolvedShots.count -
                     session.shots.filter { $0.shooter != nil }.count +
                     unresolvedShots.count)
        // Simpler: track original count via a separate @State if needed;
        // for the progress bar use resolved = original - remaining.
        let resolved = session.shots.filter { $0.shooter != nil }.count
        let totalShots = session.shots.count
        let fraction = totalShots > 0 ? Double(resolved) / Double(totalShots) : 1.0
        return VStack(spacing: 6) {
            ProgressView(value: fraction)
                .tint(.orange)
                .padding(.horizontal)
            Text("\(resolved) of \(totalShots) shots attributed")
                .font(.caption)
                .foregroundStyle(.secondary)
        }
        .padding(.bottom, 20)
    }

    // MARK: - Logic

    private func loadUnresolved() {
        unresolvedShots = session.shots.filter { $0.shooter == nil }
    }

    private func assignShot(_ shot: GameShotRecord, to player: GamePlayer) {
        shot.shooter = player
        // Update team score for makes that were not counted at detection time.
        if shot.result == .make {
            let points = shot.shotType == .threePoint ? 3 : 2
            switch player.teamAssignment {
            case .teamA: session.teamAScore += points
            case .teamB: session.teamBScore += points
            }
        }
        try? dataService.modelContext.save()
        withAnimation {
            unresolvedShots.removeAll { $0.id == shot.id }
        }
        if unresolvedShots.isEmpty { dismiss() }
    }
}
```
