```swift
// HoopTrack/Services/PlayerTracker.swift
// Phase SP2 — Tracks registered GamePlayers across live video frames using
// VNDetectHumanBodyPoseRequest. Matches detections to known players via
// AppearanceMatcher, maintains EMA-smoothed confidence, and exposes the
// current `tracked` snapshot for GameScoringCoordinator.

import Foundation
import Combine
import Vision
import CoreMedia
import CoreVideo
import CoreGraphics

// MARK: - TrackedPlayer

/// In-flight tracking state for one registered GamePlayer. Value type — the
/// PlayerTracker owns a [UUID: TrackedPlayer] dictionary and replaces entries
/// atomically each stride frame so reads on main actor are always consistent.
struct TrackedPlayer {
    let player: GamePlayer
    /// Upper-body bounding rect in normalised Vision coords (0..1, origin bottom-left).
    var lastBox: CGRect
    /// Wall clock of the last frame where this player was confidently matched.
    var lastSeen: Date
    /// EMA-smoothed match quality. 0 = lost / never seen. 1 = high confidence.
    var matchConfidence: Double

    /// Convenience: seconds since the player was last reliably detected.
    func timeSinceSeen(relativeTo now: Date = .now) -> TimeInterval {
        now.timeIntervalSince(lastSeen)
    }
}

// MARK: - PlayerTracker

/// Subscribes to CameraService.framePublisher, runs body-pose detection every
/// `HoopTrack.Game.trackerFrameStride` frames, and keeps a per-player
/// TrackedPlayer entry up to date. All mutations happen on the @MainActor;
/// Vision work is performed synchronously inside a detached Task to avoid
/// blocking the camera queue.
@MainActor
final class PlayerTracker: ObservableObject {

    // MARK: Published state

    /// Current tracking state keyed by GamePlayer.id. Read by
    /// GameScoringCoordinator on shot events.
    @Published private(set) var tracked: [UUID: TrackedPlayer] = [:]

    // MARK: Private

    private var frameCount: Int = 0
    private var cancellables = Set<AnyCancellable>()

    // MARK: Public interface

    /// Register the set of players for this session. Must be called before
    /// `subscribe(to:)`. Resets any prior tracking state.
    func configure(with players: [GamePlayer]) {
        tracked = [:]
        for player in players {
            tracked[player.id] = TrackedPlayer(
                player: player,
                lastBox: .zero,
                lastSeen: .distantPast,
                matchConfidence: 0
            )
        }
        frameCount = 0
    }

    /// Wire up the Combine pipeline to start processing frames.
    func subscribe(to framePublisher: AnyPublisher<CMSampleBuffer, Never>) {
        framePublisher
            .sink { [weak self] sampleBuffer in
                // Hop off the camera serial queue immediately — ingest is async.
                Task { @MainActor [weak self] in
                    await self?.ingest(sampleBuffer: sampleBuffer)
                }
            }
            .store(in: &cancellables)
    }

    /// Process one sample buffer. Runs the Vision request on the calling
    /// context (called from a Task so it does not block the camera queue).
    func ingest(sampleBuffer: CMSampleBuffer) async {
        frameCount += 1
        guard frameCount % HoopTrack.Game.trackerFrameStride == 0 else { return }
        guard let pixelBuffer = CMSampleBufferGetImageBuffer(sampleBuffer) else { return }
        guard !tracked.isEmpty else { return }

        // Run Vision synchronously — VNImageRequestHandler.perform is
        // thread-safe but blocking; calling from an async context is fine.
        let observations = Self.runBodyPoseRequest(on: pixelBuffer)
        guard !observations.isEmpty else {
            applyGraceWindow()
            return
        }

        // Extract descriptors for each detected body.
        var liveDescriptors: [AppearanceDescriptor] = []
        for obs in observations.prefix(6) {
            if let descriptor = extractDescriptor(from: obs, pixelBuffer: pixelBuffer) {
                liveDescriptors.append(descriptor)
            }
        }

        guard !liveDescriptors.isEmpty else {
            applyGraceWindow()
            return
        }

        // Match live detections to registered players.
        let registeredPlayers = Array(tracked.values.map(\.player))
        let matches = AppearanceMatcher.bestMatch(live: liveDescriptors, registered: registeredPlayers)

        // Build a set of matched player IDs so unmatched ones can be zeroed.
        var matchedIDs = Set<UUID>()
        let now = Date()

        for (descriptor, matchedPlayer) in matches {
            guard let matchedPlayer else { continue }
            guard var tp = tracked[matchedPlayer.id] else { continue }

            let distance = AppearanceMatcher.distance(descriptor, matchedPlayer.appearanceDescriptor ?? descriptor)
            guard distance <= HoopTrack.Game.matchAcceptThreshold else { continue }

            // Find the corresponding observation to get a bounding rect.
            if let obsIndex = liveDescriptors.firstIndex(of: descriptor),
               obsIndex < observations.count {
                let obs = observations[obsIndex]
                tp.lastBox = upperBodyBox(from: obs)
            }

            // EMA-smooth the confidence.
            let rawConfidence = max(0, 1.0 - (distance / HoopTrack.Game.matchAcceptThreshold))
            let alpha = HoopTrack.Game.playerConfidenceEMAAlpha
            tp.matchConfidence = alpha * rawConfidence + (1.0 - alpha) * tp.matchConfidence
            tp.lastSeen = now
            tracked[matchedPlayer.id] = tp
            matchedIDs.insert(matchedPlayer.id)
        }

        // Apply grace window for unmatched players.
        applyGraceWindow(except: matchedIDs)
    }

    /// Clears all tracking state and cancels the frame subscription.
    func reset() {
        tracked = [:]
        frameCount = 0
        cancellables.removeAll()
    }

    // MARK: Private helpers

    /// Zero out confidence for players that exceeded the lost-timeout window.
    private func applyGraceWindow(except matchedIDs: Set<UUID> = []) {
        let now = Date()
        for id in tracked.keys where !matchedIDs.contains(id) {
            guard var tp = tracked[id] else { continue }
            if now.timeIntervalSince(tp.lastSeen) > HoopTrack.Game.playerLostTimeoutSec {
                tp.matchConfidence = 0
                tracked[id] = tp
            }
        }
    }

    /// Compute a normalised bounding rect from the four torso keypoints of a
    /// body-pose observation. Vision coords: origin bottom-left, 0..1.
    private func upperBodyBox(from observation: VNHumanBodyPoseObservation) -> CGRect {
        let keys: [VNHumanBodyPoseObservation.JointName] = [
            .leftShoulder, .rightShoulder, .leftHip, .rightHip
        ]
        var points: [CGPoint] = []
        for key in keys {
            if let pt = try? observation.recognizedPoint(key), pt.confidence > 0 {
                points.append(pt.location)
            }
        }
        guard !points.isEmpty else { return .zero }
        let xs = points.map(\.x)
        let ys = points.map(\.y)
        let minX = xs.min()!; let maxX = xs.max()!
        let minY = ys.min()!; let maxY = ys.max()!
        return CGRect(x: minX, y: minY, width: max(maxX - minX, 0.01), height: max(maxY - minY, 0.01))
    }

    /// Extract an AppearanceDescriptor from a pose observation + pixel buffer.
    /// Delegates to AppearanceExtraction helpers to stay testable.
    private func extractDescriptor(
        from observation: VNHumanBodyPoseObservation,
        pixelBuffer: CVPixelBuffer
    ) -> AppearanceDescriptor? {
        // Gather torso keypoints.
        guard let ls = try? observation.recognizedPoint(.leftShoulder),
              let rs = try? observation.recognizedPoint(.rightShoulder),
              let lh = try? observation.recognizedPoint(.leftHip),
              let rh = try? observation.recognizedPoint(.rightHip),
              ls.confidence > 0, rs.confidence > 0,
              lh.confidence > 0, rh.confidence > 0
        else { return nil }

        let torsoBox = AppearanceExtraction.upperBodyBox(
            leftShoulder: ls.location,
            rightShoulder: rs.location,
            leftHip: lh.location,
            rightHip: rh.location
        )

        // Convert normalised Vision rect to pixel-space for histogram extraction.
        let width  = CGFloat(CVPixelBufferGetWidth(pixelBuffer))
        let height = CGFloat(CVPixelBufferGetHeight(pixelBuffer))
        // Vision origin is bottom-left; CIImage origin is also bottom-left — no flip needed.
        let pixelRect = CGRect(
            x: torsoBox.minX * width,
            y: torsoBox.minY * height,
            width: torsoBox.width * width,
            height: torsoBox.height * height
        )

        let ciImage = CIImage(cvPixelBuffer: pixelBuffer)
        let (hue, value) = AppearanceExtraction.histograms(from: ciImage, rect: pixelRect)

        // Height ratio: nose to ankle.
        var heightRatioValue: Float = 0
        if let nose   = try? observation.recognizedPoint(.nose),
           let lAnkle = try? observation.recognizedPoint(.leftAnkle),
           let rAnkle = try? observation.recognizedPoint(.rightAnkle),
           nose.confidence > 0 {
            heightRatioValue = AppearanceExtraction.heightRatio(
                nose: nose.location,
                leftAnkle: lAnkle.location,
                rightAnkle: rAnkle.location
            )
        }

        let aspect: Float = torsoBox.height > 0
            ? Float(torsoBox.width / torsoBox.height)
            : 1.0

        return AppearanceDescriptor(
            torsoHueHistogram: hue,
            torsoValueHistogram: value,
            heightRatio: heightRatioValue,
            upperBodyAspect: aspect,
            schemaVersion: HoopTrack.Game.appearanceDescriptorSchemaVersion
        )
    }

    /// Run VNDetectHumanBodyPoseRequest synchronously on a pixel buffer.
    /// Returns at most 6 observations (matches the Vision default max).
    private static func runBodyPoseRequest(on pixelBuffer: CVPixelBuffer) -> [VNHumanBodyPoseObservation] {
        let request = VNDetectHumanBodyPoseRequest()
        let handler = VNImageRequestHandler(
            cvPixelBuffer: pixelBuffer,
            orientation: .up,
            options: [:]
        )
        do {
            try handler.perform([request])
        } catch {
            return []
        }
        return (request.results ?? []).prefix(6).map { $0 }
    }
}
```

```swift
// HoopTrack/Services/GameScoringCoordinator.swift
// Phase SP2 — Listens to CVPipeline.shotPublisher, attributes each ShotEvent
// to the nearest registered GamePlayer via PlayerTracker, persists the shot
// via DataService.addGameShot, and publishes a KillfeedEntry for the live HUD.

import Foundation
import Combine
import CoreGraphics

// MARK: - KillfeedEntry

/// One line in the live shot-attribution feed shown during Game Mode.
struct KillfeedEntry: Identifiable, Equatable {
    let id: UUID
    let playerName: String
    let teamAssignment: TeamAssignment
    let shotType: GameShotType
    let result: ShotResult
    let timestamp: Date
}

// MARK: - GameScoringCoordinator

@MainActor
final class GameScoringCoordinator: ObservableObject {

    // MARK: Published state

    /// Newest entry at index 0. Capped at `HoopTrack.Game.killfeedMaxEntries`.
    @Published private(set) var killfeedEntries: [KillfeedEntry] = []

    // MARK: Dependencies

    private let cvPipeline: CVPipeline
    private let playerTracker: PlayerTracker
    private let session: GameSession
    private let dataService: DataService

    // MARK: Private

    private var cancellables = Set<AnyCancellable>()

    // MARK: Init

    init(
        cvPipeline: CVPipeline,
        playerTracker: PlayerTracker,
        session: GameSession,
        dataService: DataService
    ) {
        self.cvPipeline = cvPipeline
        self.playerTracker = playerTracker
        self.session = session
        self.dataService = dataService
    }

    // MARK: Public interface

    /// Start listening for shot events from CVPipeline.
    func start() {
        cvPipeline.shotPublisher
            .receive(on: DispatchQueue.main)
            .sink { [weak self] event in
                Task { @MainActor [weak self] in
                    self?.handleShotEvent(event)
                }
            }
            .store(in: &cancellables)
    }

    /// Cancel the subscription and reset the killfeed.
    func stop() {
        cancellables.removeAll()
        killfeedEntries = []
    }

    // MARK: Private — attribution

    private func handleShotEvent(_ event: ShotEvent) {
        let now = Date()
        let candidates = Array(playerTracker.tracked.values)

        // Find the best-attributed player using AttributionGating.
        var bestPlayer: TrackedPlayer?
        var bestConfidence: Double = 0

        for candidate in candidates {
            let timeSinceSeen = now.timeIntervalSince(candidate.lastSeen)
            guard AttributionGating.shouldAttribute(
                releasePoint: event.releasePoint,
                trackedBox: candidate.lastBox,
                timeSinceSeen: timeSinceSeen
            ) else { continue }

            let confidence = attributionConfidence(
                for: candidate,
                releasePoint: event.releasePoint,
                timeSinceSeen: timeSinceSeen
            )
            if confidence > bestConfidence {
                bestConfidence = confidence
                bestPlayer = candidate
            }
        }

        let shotType = classify(event)
        let shooter = bestPlayer?.player
        let finalConfidence = bestPlayer != nil ? bestConfidence : 0.0

        // Persist via DataService (throws are swallowed — CV pipeline should not crash on persistence errors).
        try? dataService.addGameShot(
            to: session,
            shooter: shooter,
            result: event.result,
            courtX: event.courtX,
            courtY: event.courtY,
            shotType: shotType,
            attributionConfidence: finalConfidence
        )

        // Publish killfeed entry only for attributed shots with known players.
        if let player = shooter {
            let entry = KillfeedEntry(
                id: UUID(),
                playerName: player.name,
                teamAssignment: player.teamAssignment,
                shotType: shotType,
                result: event.result,
                timestamp: now
            )
            killfeedEntries.insert(entry, at: 0)
            if killfeedEntries.count > HoopTrack.Game.killfeedMaxEntries {
                killfeedEntries.removeLast(killfeedEntries.count - HoopTrack.Game.killfeedMaxEntries)
            }
        }
    }

    // MARK: Private — helpers

    /// Composite attribution confidence: tracker EMA × distance factor × recency factor.
    private func attributionConfidence(
        for candidate: TrackedPlayer,
        releasePoint: CGPoint,
        timeSinceSeen: TimeInterval
    ) -> Double {
        let trackerConfidence = candidate.matchConfidence

        // Distance factor: how close is the release point to the player's last box.
        let distance = releasePoint.distance(to: candidate.lastBox.center)
        let distanceFactor = max(0, 1.0 - (distance / HoopTrack.Game.attributionReleaseDistMax))

        // Recency factor: how recently this player was seen.
        let recencyFactor = max(0, 1.0 - (timeSinceSeen / HoopTrack.Game.attributionTimeSinceSeenMax))

        return trackerConfidence * distanceFactor * recencyFactor
    }

    /// Map a ShotEvent zone to GameShotType for scoring.
    private func classify(_ event: ShotEvent) -> GameShotType {
        switch event.zone {
        case .cornerThree, .aboveBreakThree:
            return .threePoint
        default:
            return .twoPoint
        }
    }
}

// MARK: - CGPoint distance helper (private to this file)

private extension CGPoint {
    func distance(to other: CGPoint) -> Double {
        let dx = x - other.x
        let dy = y - other.y
        return Double((dx * dx + dy * dy).squareRoot())
    }
}

// MARK: - CGRect center helper (private to this file)

private extension CGRect {
    var center: CGPoint {
        CGPoint(x: midX, y: midY)
    }
}
```

```swift
// GameSessionViewModel.swift
// SP1: shell that owns the current GameSession row and exposes manual
// make/miss helpers for LiveGameView. SP2 adds optional CV mode driven by
// GameScoringCoordinator + PlayerTracker — activated when a CVPipeline and
// CameraService frame publisher are provided.

import Foundation
import Combine
import AVFoundation

@MainActor
final class GameSessionViewModel: ObservableObject {

    // MARK: - Published state

    @Published private(set) var session: GameSession
    /// True when CV attribution is running. Gates CV overlay visibility in the UI.
    @Published var cvModeActive: Bool = false

    // MARK: - Dependencies

    private let dataService: DataService

    // MARK: - SP2: CV services (optional — nil when running in manual mode)

    private(set) var scoringCoordinator: GameScoringCoordinator?
    private(set) var playerTracker: PlayerTracker?

    // MARK: - Init

    init(session: GameSession, dataService: DataService) {
        self.session = session
        self.dataService = dataService
    }

    // MARK: - SP1: Manual shot logging (still used when CV mode is off)

    /// Manual make/miss for SP1 and as a fallback when cvModeActive == false.
    func logShot(
        shooter: GamePlayer,
        result: ShotResult,
        courtX: Double,
        courtY: Double,
        shotType: GameShotType
    ) {
        try? dataService.addGameShot(
            to: session,
            shooter: shooter,
            result: result,
            courtX: courtX, courtY: courtY,
            shotType: shotType
        )
        objectWillChange.send()
    }

    // MARK: - SP2: CV mode activation

    /// Start CV-powered attribution. Creates a PlayerTracker and
    /// GameScoringCoordinator, wires them together, and begins processing
    /// frames from `framePublisher`.
    ///
    /// Call this after the session transitions to `.inProgress` and the
    /// CVPipeline has been started by the parent view.
    func startCVMode(
        cvPipeline: CVPipeline,
        framePublisher: AnyPublisher<CMSampleBuffer, Never>,
        session: GameSession
    ) {
        let tracker = PlayerTracker()
        tracker.configure(with: session.players)
        tracker.subscribe(to: framePublisher)

        let coordinator = GameScoringCoordinator(
            cvPipeline: cvPipeline,
            playerTracker: tracker,
            session: session,
            dataService: dataService
        )
        coordinator.start()

        self.playerTracker = tracker
        self.scoringCoordinator = coordinator
        self.cvModeActive = true
    }

    /// Stop CV attribution — call when the session ends or the view disappears.
    func stopCVMode() {
        scoringCoordinator?.stop()
        playerTracker?.reset()
        scoringCoordinator = nil
        playerTracker = nil
        cvModeActive = false
    }

    // MARK: - SP2: Killfeed passthrough

    /// Live shot-attribution feed for the in-game HUD. Empty when CV mode is off.
    var killfeedEntries: [KillfeedEntry] {
        scoringCoordinator?.killfeedEntries ?? []
    }

    // MARK: - Session lifecycle

    func endSession() {
        stopCVMode()
        session.gameState = .completed
        session.endTimestamp = .now
        try? dataService.modelContext.save()
    }
}
```

```swift
// HoopTrackTests/PlayerTrackerTests.swift
// Phase SP2 — Unit tests for PlayerTracker configuration and reset logic.
// Tests use an in-memory ModelContainer so no on-disk SwiftData state is written.

import XCTest
import SwiftData
@testable import HoopTrack

@MainActor
final class PlayerTrackerTests: XCTestCase {

    // MARK: - Helpers

    /// Returns an in-memory ModelContainer + context for building GamePlayer test fixtures.
    private func makeContext() throws -> ModelContext {
        let config = ModelConfiguration(isStoredInMemoryOnly: true)
        let container = try ModelContainer(for: GamePlayer.self, configurations: config)
        return ModelContext(container)
    }

    /// Build a minimal valid AppearanceDescriptor encoded as Data.
    private func makeEmbeddingData() throws -> Data {
        let hueBins   = HoopTrack.Game.histogramHueBins
        let valueBins = HoopTrack.Game.histogramValueBins
        let descriptor = AppearanceDescriptor(
            torsoHueHistogram:   Array(repeating: Float(1.0 / Float(hueBins)), count: hueBins),
            torsoValueHistogram: Array(repeating: Float(1.0 / Float(valueBins)), count: valueBins),
            heightRatio:   0.5,
            upperBodyAspect: 0.6,
            schemaVersion: HoopTrack.Game.appearanceDescriptorSchemaVersion
        )
        return try JSONEncoder().encode(descriptor)
    }

    // MARK: - configure tests

    /// configure(with:) should create one TrackedPlayer per player with zero
    /// matchConfidence and a lastSeen date in the distant past.
    func test_configure_populatesTracked() throws {
        let context = try makeContext()
        let embeddingData = try makeEmbeddingData()

        let player1 = GamePlayer(name: "Alice", appearanceEmbedding: embeddingData, teamAssignment: .teamA)
        let player2 = GamePlayer(name: "Bob",   appearanceEmbedding: embeddingData, teamAssignment: .teamB)
        context.insert(player1)
        context.insert(player2)

        let tracker = PlayerTracker()
        tracker.configure(with: [player1, player2])

        XCTAssertEqual(tracker.tracked.count, 2, "tracked should contain one entry per registered player")

        for (_, tp) in tracker.tracked {
            XCTAssertEqual(tp.matchConfidence, 0,
                "Initial matchConfidence must be 0 — no frames ingested yet")
            XCTAssertEqual(tp.lastBox, .zero,
                "Initial lastBox must be .zero — no pose observed yet")
            XCTAssertTrue(
                tp.lastSeen < Date(timeIntervalSinceNow: -60),
                "Initial lastSeen should be .distantPast (well before now)"
            )
        }
    }

    /// configure(with:) called a second time should replace the previous state
    /// rather than merging — this covers the re-registration path.
    func test_configure_replacesExistingState() throws {
        let context = try makeContext()
        let embeddingData = try makeEmbeddingData()

        let playerA = GamePlayer(name: "Alice", appearanceEmbedding: embeddingData, teamAssignment: .teamA)
        let playerB = GamePlayer(name: "Bob",   appearanceEmbedding: embeddingData, teamAssignment: .teamB)
        let playerC = GamePlayer(name: "Carol", appearanceEmbedding: embeddingData, teamAssignment: .teamA)
        context.insert(playerA)
        context.insert(playerB)
        context.insert(playerC)

        let tracker = PlayerTracker()
        tracker.configure(with: [playerA, playerB])
        XCTAssertEqual(tracker.tracked.count, 2)

        tracker.configure(with: [playerC])
        XCTAssertEqual(tracker.tracked.count, 1,
            "Re-configure should replace tracked entirely with the new player list")
        XCTAssertNotNil(tracker.tracked[playerC.id])
        XCTAssertNil(tracker.tracked[playerA.id])
        XCTAssertNil(tracker.tracked[playerB.id])
    }

    // MARK: - reset tests

    /// reset() should clear all tracked state and stop the frame count.
    func test_reset_clearsTracked() throws {
        let context = try makeContext()
        let embeddingData = try makeEmbeddingData()

        let player = GamePlayer(name: "Alice", appearanceEmbedding: embeddingData, teamAssignment: .teamA)
        context.insert(player)

        let tracker = PlayerTracker()
        tracker.configure(with: [player])
        XCTAssertEqual(tracker.tracked.count, 1, "Precondition: tracked populated after configure")

        tracker.reset()
        XCTAssertTrue(tracker.tracked.isEmpty, "reset() must clear the tracked dictionary")
    }

    /// After reset(), a subsequent configure() should work correctly — verifying
    /// the frame counter is also zeroed.
    func test_reset_thenConfigureWorks() throws {
        let context = try makeContext()
        let embeddingData = try makeEmbeddingData()

        let player = GamePlayer(name: "Alice", appearanceEmbedding: embeddingData, teamAssignment: .teamA)
        context.insert(player)

        let tracker = PlayerTracker()
        tracker.configure(with: [player])
        tracker.reset()
        tracker.configure(with: [player])

        XCTAssertEqual(tracker.tracked.count, 1,
            "configure after reset should repopulate tracked cleanly")
        XCTAssertEqual(tracker.tracked[player.id]?.matchConfidence, 0)
    }
}
```
