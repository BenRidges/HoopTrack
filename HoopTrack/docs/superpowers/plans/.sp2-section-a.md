```swift
// HoopTrack/Utilities/AppearanceMatcher.swift
// SP2 — Pure appearance-descriptor distance math for player attribution.
// No Vision, no AVFoundation — unit-testable in isolation.

import Foundation

enum AppearanceMatcher {

    // MARK: - Chi-Squared Distance

    /// Symmetric chi-squared distance between two normalised histograms.
    /// Returns a value in [0, ∞); caller is responsible for clamping if needed.
    ///
    /// Formula: Σᵢ (Aᵢ - Bᵢ)² / (Aᵢ + Bᵢ + ε),  ε = 1e-10
    static func chiSquared(_ a: [Float], _ b: [Float]) -> Double {
        precondition(a.count == b.count, "Histogram bin counts must match")
        let epsilon: Double = 1e-10
        return zip(a, b).reduce(0.0) { acc, pair in
            let ai = Double(pair.0)
            let bi = Double(pair.1)
            let diff = ai - bi
            return acc + (diff * diff) / (ai + bi + epsilon)
        }
    }

    // MARK: - Weighted Distance

    /// Weighted appearance distance between two `AppearanceDescriptor`s.
    ///
    /// Returns 1.0 (worst case) immediately if either descriptor fails `isWellFormed`.
    ///
    /// Weights:
    /// - Hue chi-squared   × 0.5
    /// - Value chi-squared × 0.2
    /// - |heightRatio|     × 0.2
    /// - |upperBodyAspect| × 0.1
    ///
    /// Result is clamped to [0, 1].
    static func distance(_ a: AppearanceDescriptor, _ b: AppearanceDescriptor) -> Double {
        guard a.isWellFormed, b.isWellFormed else { return 1.0 }

        let hueDist    = chiSquared(a.torsoHueHistogram, b.torsoHueHistogram)
        let valueDist  = chiSquared(a.torsoValueHistogram, b.torsoValueHistogram)
        let heightDiff = abs(Double(a.heightRatio) - Double(b.heightRatio))
        let aspectDiff = abs(Double(a.upperBodyAspect) - Double(b.upperBodyAspect))

        let raw = hueDist   * 0.5
              + valueDist  * 0.2
              + heightDiff * 0.2
              + aspectDiff * 0.1

        return min(max(raw, 0.0), 1.0)
    }
}
```

```swift
    // MARK: - Greedy Best-Match Assignment

    /// Greedy one-to-one assignment of live descriptors to registered players.
    ///
    /// For each live descriptor (in input order), finds the registered `GamePlayer`
    /// whose `appearanceDescriptor` has the lowest `distance(_:_:)` below
    /// `HoopTrack.Game.matchAcceptThreshold`. Once a player is claimed, it is
    /// excluded from subsequent assignments (one-to-one constraint).
    ///
    /// - Parameters:
    ///   - live: Array of freshly-computed `AppearanceDescriptor`s from the current frame.
    ///   - registered: The registered `GamePlayer`s to match against.
    /// - Returns: Parallel array to `live`. Each element is `(liveDescriptor, matchedPlayer?)`.
    ///            `nil` means no registered player was close enough.
    static func bestMatch(
        live: [AppearanceDescriptor],
        registered: [GamePlayer]
    ) -> [(AppearanceDescriptor, GamePlayer?)] {
        let threshold = HoopTrack.Game.matchAcceptThreshold
        var available = registered  // shrinks as players are claimed

        return live.map { liveDesc in
            // Find best (lowest-distance) available player under threshold.
            var bestDist = threshold
            var bestIdx: Int? = nil

            for (idx, player) in available.enumerated() {
                guard let regDesc = player.appearanceDescriptor else { continue }
                let d = distance(liveDesc, regDesc)
                if d < bestDist {
                    bestDist = d
                    bestIdx  = idx
                }
            }

            if let idx = bestIdx {
                let matched = available.remove(at: idx)
                return (liveDesc, matched)
            } else {
                return (liveDesc, nil)
            }
        }
    }
```

```swift
// No Vision, no AVFoundation, no SwiftData — fully unit-testable.

import CoreGraphics
import Foundation

/// Decides whether a detected shot can be attributed to a specific tracked player.
///
/// Both `releasePoint` and `trackedBox` use normalised frame coordinates (0…1).
/// Both thresholds are exclusive (strict `<`).
enum AttributionGating {

    /// Returns `true` when a shot at `releasePoint` is close enough to `trackedBox`
    /// and the player was seen recently enough to warrant attribution.
    ///
    /// - Parameters:
    ///   - releasePoint:  Normalised (0…1) frame coordinate of the ball at release.
    ///   - trackedBox:    Normalised (0…1) bounding rect of the tracked player body.
    ///   - timeSinceSeen: Seconds elapsed since the player's tracking was last updated.
    /// - Returns: `true` iff `distance < attributionReleaseDistMax`
    ///            **and** `timeSinceSeen < attributionTimeSinceSeenMax`.
    static func shouldAttribute(
        releasePoint:  CGPoint,
        trackedBox:    CGRect,
        timeSinceSeen: TimeInterval
    ) -> Bool {
        let boxCentre = CGPoint(x: trackedBox.midX, y: trackedBox.midY)
        let dx        = releasePoint.x - boxCentre.x
        let dy        = releasePoint.y - boxCentre.y
        let distance  = sqrt(Double(dx * dx + dy * dy))

        return distance    < HoopTrack.Game.attributionReleaseDistMax
            && timeSinceSeen < HoopTrack.Game.attributionTimeSinceSeenMax
    }
}
```
